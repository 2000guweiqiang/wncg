def sxh(start, end):
    if end >= start and start >=100 and end<= 999:
        num=start 
        L =[]
        while num<=end:
            x=num%10      
            y=num//10%10  
            z=num//100    
            if (num==pow(x,3)+pow(y,3)+pow(z,3)):
                L.append(num) 
            num=num+1
        return L   
print(sxh(100,999))
